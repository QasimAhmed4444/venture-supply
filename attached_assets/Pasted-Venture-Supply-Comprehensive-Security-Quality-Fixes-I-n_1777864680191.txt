tsimport { getSupabase } from "./supabase.js";

const VAT_RATE = 0.15;
const FREE_DELIVERY_THRESHOLD = 200;
const DELIVERY_CHARGE = 25;

type OrderInput = {
  customerId: string | null;
  customerType: "b2c" | "b2b" | "guest";
  items: Array<{ productId: string; quantity: number; packId?: string | null }>;
  couponCode?: string | null;
  paymentMethod: "cod" | "bank" | "credit";
};

export type PricedOrder = {
  items: Array<{
    productId: string;
    productName: string;
    quantity: number;
    unitPrice: number;
    lineTotal: number;
    packId: string | null;
  }>;
  subtotal: number;
  vat: number;
  delivery_charge: number;
  discount: number;
  coupon_code: string | null;
  total: number;
};

export async function priceOrder(input: OrderInput): Promise<PricedOrder> {
  const sb = getSupabase();
  if (!sb) throw new Error("Database unavailable");
  if (!input.items || input.items.length === 0) throw new Error("Empty cart");
  if (input.items.length > 50) throw new Error("Too many items (max 50 per order)");

  const productIds = input.items.map((i) => i.productId);
  const { data: products, error: prodErr } = await sb
    .from("products")
    .select("id, en_name, b2c_price, b2b_price, audience, stock_qty, stock_status")
    .in("id", productIds);
  if (prodErr) throw new Error("Product fetch failed: " + prodErr.message);

  const productMap = new Map(products?.map((p) => [p.id, p]) ?? []);
  const pricedItems: PricedOrder["items"] = [];

  for (const item of input.items) {
    const p = productMap.get(item.productId);
    if (!p) throw new Error(`Product ${item.productId} not found`);
    if (p.stock_status === "out-of-stock") throw new Error(`${p.en_name} is out of stock`);
    if (item.quantity <= 0 || !Number.isInteger(item.quantity)) throw new Error("Invalid quantity");
    if (item.quantity > Number(p.stock_qty ?? 0)) {
      throw new Error(`Only ${p.stock_qty} of ${p.en_name} available`);
    }
    if (input.customerType === "b2c" && p.audience === "b2b") {
      throw new Error(`${p.en_name} is B2B-only`);
    }
    if (input.customerType === "b2b" && p.audience === "b2c") {
      throw new Error(`${p.en_name} is B2C-only`);
    }

    const unitPrice =
      input.customerType === "b2b" && Number(p.b2b_price) > 0
        ? Number(p.b2b_price)
        : Number(p.b2c_price);
    const lineTotal = Math.round(unitPrice * item.quantity * 100) / 100;

    pricedItems.push({
      productId: p.id,
      productName: p.en_name,
      quantity: item.quantity,
      unitPrice,
      lineTotal,
      packId: item.packId ?? null,
    });
  }

  const subtotal = Math.round(pricedItems.reduce((s, i) => s + i.lineTotal, 0) * 100) / 100;

  let discount = 0;
  let appliedCouponCode: string | null = null;
  if (input.couponCode) {
    const code = String(input.couponCode).trim().toUpperCase();
    const { data: coupon } = await sb
      .from("coupons")
      .select("code, type, value, min_order, audience, max_uses, uses_count, is_active, starts_at, ends_at")
      .ilike("code", code)
      .single();
    if (!coupon) throw new Error("Invalid coupon");
    if (!coupon.is_active) throw new Error("Coupon inactive");
    if (coupon.audience !== "both" && coupon.audience !== input.customerType) {
      throw new Error("Coupon not applicable to your account type");
    }
    if (coupon.starts_at && new Date(coupon.starts_at) > new Date()) {
      throw new Error("Coupon not yet active");
    }
    if (coupon.ends_at && new Date(coupon.ends_at) < new Date()) {
      throw new Error("Coupon expired");
    }
    if (coupon.min_order && subtotal < Number(coupon.min_order)) {
      throw new Error(`Minimum order ${coupon.min_order} SAR for this coupon`);
    }
    if (coupon.max_uses && coupon.uses_count >= coupon.max_uses) {
      throw new Error("Coupon redemption limit reached");
    }

    if (coupon.type === "percent") {
      discount = Math.round((subtotal * Number(coupon.value)) / 100 * 100) / 100;
    } else if (coupon.type === "fixed") {
      discount = Math.min(Number(coupon.value), subtotal);
    }
    appliedCouponCode = coupon.code;
  }

  const taxableAmount = Math.max(0, subtotal - discount);
  const vat = Math.round(taxableAmount * VAT_RATE * 100) / 100;
  const delivery_charge = subtotal >= FREE_DELIVERY_THRESHOLD ? 0 : DELIVERY_CHARGE;
  const total = Math.round((taxableAmount + vat + delivery_charge) * 100) / 100;

  return { items: pricedItems, subtotal, vat, delivery_charge, discount, coupon_code: appliedCouponCode, total };
}

export async function incrementCouponUses(code: string | null): Promise<boolean> {
  const sb = getSupabase();
  if (!sb || !code) return true;
  const { data, error } = await sb.rpc("increment_coupon_uses", { p_code: code });
  if (error) {
    console.error("Coupon increment failed:", error);
    return false;
  }
  return data === true;
}

export async function decrementCouponUses(code: string | null): Promise<void> {
  const sb = getSupabase();
  if (!sb || !code) return;
  await sb.rpc("decrement_coupon_uses", { p_code: code });
}

tsimport crypto from "crypto";
import bcrypt from "bcryptjs";
// ... assuming authLimiter is imported from app.ts

router.post("/auth/request-password-reset", authLimiter, async (req, res) => {
  const sb = getSupabase();
  if (!sb) return res.status(503).json({ error: "Database unavailable" });

  const email = String(req.body?.email ?? "").trim().toLowerCase();
  if (!email) return res.status(400).json({ error: "Email required" });

  const { data: user } = await sb.from("staff").select("id, email").eq("email", email).maybeSingle();

  if (user) {
    const token = crypto.randomBytes(32).toString("hex");
    const tokenHash = crypto.createHash("sha256").update(token).digest("hex");
    const expiresAt = new Date(Date.now() + 60 * 60 * 1000).toISOString();

    await sb.from("staff").update({
      reset_token_hash: tokenHash,
      reset_token_expires_at: expiresAt,
    }).eq("id", user.id);

    // TODO: send email with reset link. For now, log it.
    const resetUrl = `${process.env.APP_URL ?? "http://localhost:3000"}/reset-password?token=${token}`;
    console.log(`[RESET] Password reset link for ${user.email}: ${resetUrl}`);
    // Wire to email provider (Resend, SendGrid, etc.) when EMAIL_API_KEY is configured.
  }

  // Always return 200 to prevent email enumeration

router.post("/auth/reset-password", authLimiter, async (req, res) => {
  const sb = getSupabase();
  if (!sb) return res.status(503).json({ error: "Database unavailable" });

  const { token, newPassword } = req.body ?? {};
  if (!token || !newPassword) return res.status(400).json({ error: "Token and newPassword required" });
  if (String(newPassword).length < 8) return res.status(400).json({ error: "Password must be 8+ characters" });

  const tokenHash = crypto.createHash("sha256").update(String(token)).digest("hex");

  const { data: user } = await sb
    .from("staff")
    .select("id, reset_token_expires_at")
    .eq("reset_token_hash", tokenHash)
    .maybeSingle();

  if (!user) return res.status(400).json({ error: "Invalid or expired token" });
  if (!user.reset_token_expires_at || new Date(user.reset_token_expires_at) < new Date()) {
    return res.status(400).json({ error: "Invalid or expired token" });
  }

  const newHash = await bcrypt.hash(String(newPassword), 12);

  await sb.from("staff").update({
    password: newHash,
    reset_token_hash: null,
    reset_token_expires_at: null,
  }).eq("id", user.id);

  return res.json({ ok: true });
});
